bool findFunctionDefinition(const Text programText, const char targetFuncName[MAX_LINE_LENGTH + 1], Text* funcDefinition)
{
    Считать, что искомая функция не найдена
	Обнулить выходной параметр с определением искомой функии

    Для каждой строки текста, пока не найдена искомая функция
    {
        Если отсутсвуют строки в выходном параметре с определением искомой функции 
        {
            Если название функции имеется в строке и 
            название функции является началом определения функции
            {
                Записать строку в выходной параметр с определением искомой функции
                Увеличить счетчик количества строк с определением искомой фукнции
            }
        } 
        иначе 
        {
            Записать строку выходной параметр с определением искомой функции
            Увеличить счетчик количества строк с определением искомой фукнции
            Если строка содержит конец определения функции
            {
                Считать что искомая функция найдена
            }
        }
    }
}
